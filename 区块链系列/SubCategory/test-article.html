class Test {
  private $name;
  private $age;
  public function __construct($array) {
    $this->name = $array['name'];
    $this->age = $array['age'];
  }

  public function getName() {
    return $this->name;
  }

  public function setName($name) {
    $this->name = $name;
  }

  public function getAge() {
    return $this->age;
  }

  public function show() {
    printf('my name is %s,age is %d',$this->name,$this->age);
  }

}
